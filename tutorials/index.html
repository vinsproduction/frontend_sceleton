Модальные окна (Popups)

Инициализация (js)

$(document).ready(function() {
 window.popup2 = new Popup({
  group: 'group_name',
  top: 50,
  left: 150,
  scroll: false,
 });
 window.popup3 = new Popup({
  fade: false,
 });
 popup2.doCallback = function(popup) {
  console.log('this doCallback. and this popup2 name:', popup.attr('data-popup-name'));
 }
 popup2.loadCallback = function(popup) {
  console.log('this loadCallback. and this popup2 name:', popup.attr('data-popup-name'));
 }
 popup.doCallback = function(popup) {
  console.log('this doCallback. and this popup name:', popup.attr('data-popup-name'));
 }
 popup.loadCallback = function(popup) {
  console.log('this loadCallback. and this popup name:', popup.attr('data-popup-name'));
 }
})
